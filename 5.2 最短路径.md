## Dijkstra
* 特点
> 适用于权值为非负的图的单源最短路径，用斐波那契堆的复杂度O(E+VlgV)

* 无权值的模板代码
```c++
# 不可达距离
#define INF 1<<29

# n 点的数量； b 开始点
int n, b;
# 数据 
int data[85][85];
# 最短路径的路径
int pre[85];
# 记录点是否被访问
bool vis[85] = {false};
# 记录开始点到各点的最短距离
int dst[85];

void dijkstra() {
    fill(dst, dst + n, INF);
    dst[b-1] = 0;

    for (int i = 0; i < n; ++i) {
        int min = INF, t = -1;
        for (int j = 0; j < n; ++j) {
            if (!vis[j] && dst[j] < min) {
                min = dst[j];
                t = j;
            }
        }

        if (t == -1) return;
        vis[t] = true;

        for (int j = 0; j < n; ++j) {
            if (!vis[j] && data[t][j] != INF && data[t][j] + dst[t] < dst[j]) {
                dst[j] = data[t][j] + dst[t];
                pre[j] = t;
            }
        }
    }
}
```

* Dijkstra + DFS
```c++
int n, m;
int b, e;
int minCost = INF;
int data[1005][1005];
int cost[1005][1005];
bool vis[1005] = {false};
int dst[1005];
vector<int> pre[1005];
vector<int> tempPath, path;

void dijkstra() {
    fill(dst, dst + 1005, INF);
    fill(vis, vis + 1005, false);
    dst[b] = 0;
    for (int i = 1; i <= n; ++i) {
        int min = INF, t = -1;
        for (int j = 1; j <= n; ++j) {
            if (!vis[j] && dst[j] < min) {
                min = dst[j];
                t = j;
            }
        }
 
        if (t == -1) return;
        vis[t] = true;
        for (int j = 1; j <= n; ++j) {
            if (!vis[j] && data[t][j] != INF) {
                if (data[t][j] + dst[t] < dst[j]) {
                    dst[j] = data[t][j] + dst[t];
                    pre[j].clear();
                    pre[j].push_back(t);
                } else if (data[t][j] + dst[t] == dst[j]) {
                    pre[j].push_back(t);
                }
            }
        }
    }
 
}
 
void dfs(int end) {
    if (end == b) {
        tempPath.push_back(end);
        int tempCost = 0;
        for (int i = tempPath.size() - 1; i > 0; --i) {
            int id = tempPath[i], id_next = tempPath[i - 1];
            tempCost += cost[id][id_next];
        }
        if (tempCost < minCost) {
            minCost = tempCost;
            path = tempPath;
        }
        tempPath.pop_back();
        return;
    }
    tempPath.push_back(end);
    for (int i = 0; i < pre[end].size(); ++i) {
        dfs(pre[end][i]);
    }
    tempPath.pop_back();
}

int main(){
    while (cin >> n >> m && n != 0 && m != 0) {
 
        fill(data[0], data[0] + 1005 * 1005, INF);
        fill(cost[0], cost[0] + 1005 * 1005, INF);
 
        for (int i = 0; i < m; ++i) {
            int x, y, a, b;
            cin >> x >> y >> a >> b;
            data[x][y] = data[y][x] = a;
            cost[x][y] = cost[y][x] = b;
        }
        cin >> b >> e;
 
        dijkstra();
        dfs(e);
         
        printf("%d %d\n", dst[e], minCost);
        minCost = INF;
        for (int i = 0; i < n; ++i) {
            pre[i].clear();
        }
        tempPath.clear();
        path.clear();
    }    

    return 0;
}
```

## Bellman-Ford 和 SPFA
* 特点
> Bellman-Ford: 适用于权值有负值的图的单源最短路径，并且能够检测负圈，复杂度O(VE)  
> SPFA: 适用于权值有负值，且没有负圈的图的单源最短路径，论文中的复杂度O(kE)，k为每个节点进入Queue的次数，且k一般<=2，但此处的复杂度证明是有问题的，其实SPFA的最坏情况应该是O(VE).

* Bellman-Ford

```c++
vector<node> data[maxn];
int n, dst[maxn], num[maxn];
bool vis[maxn];

bool SPFA(int s){
    
    memset(vis, false, sizeof(vis));
    memset(num, 0, sizeof(num));
    fill(dst, dst + maxn, INF);
    
    queue<int> q;
    q.push(s);
    vis[s] = true;
    num[s]++;
    dst[s] = 0;
    while (!q.empty()){
        int u = q.front();
        q.pop();
        vis[u] = false;
        for (int j = 0; j < data[u].size(); ++j) {
            int v = data[u][j].v;
            int dis = data[u][j].dis;
            if(dst[u] + dis < dst[v]){
                dst[v] = dst[u] + dis;
                if(!vis[v]){
                    q.push(v);
                    vis[v] = tree;
                    num[v]++;
                    if(num[v] >= n)
                        return false;
                }
            }
        }
        
    }
    return true;
}
```

## Floyd

* 特点
> 每对节点之间的最短路径。


* 模板代码
```c++
void Floyd(){
    for (int k = 0; k < n; ++k) {
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                if(data[i][k] != INF && data[k][j] !=INF && data[i][k] + data[k][j] < data[i][j]){
                    data[i][j] = data[i][k] + data[k][j];
                }
            }
        }
    }
}
```
