## 定义二叉树
* 动态实现
```c++
struct node{
    typename data;
    node * lchild;
    node * rchild;
};

node* root = NULL;

// 创建新节点
node* newNode(int data){
    node* root = new node;
    root->data = data;
    root->lchild = root->rchild = NULL;
    return root;
}

// 更新节点

    if(root->data < data){
        insert(root->lchild, data);

// 二叉树的创建
node* Create(int data[], int n){
    node* root = NULL;
    for(int i=0;i < n;i++){
        insert(root, data[i]);
    }
    return root;
}

```

## 二叉树的遍历
* 先序遍历
```c++
void preOrder(node* root){
    if(root == NULL){
        return;
    }
    printf("%d\n", root->data);
    preOrder(root->lchild);
    preOrder(root->rchild);
}

```

* 中序遍历
```c++
void inOrder(node* root){
    if(root == NULL){
        return;
    }
   
    preOrder(root->lchild);
    printf("%d\n", root->data);
    preOrder(root->rchild);
}

```
* 后序遍历
```c++
void inOrder(node* root){
    if(root == NULL){
        return;
    }
   
    preOrder(root->lchild);
    preOrder(root->rchild);
    printf("%d\n", root->data);
}
```

* 层序遍历
```c++
void layerOrder(node* root){
    queue<node*> q;
    q.push(root);
    while(!q.empty()){
        node* now = q.front();
        q.pop();
        printf("%d",now->data);
        if(now->lchild != NULL) q.push(now->lchild);
        if(now->rchild != NULL) q.push(now->rchild);
    }
}
```
